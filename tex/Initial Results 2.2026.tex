\section{Data and Measurement}

\subsection{Data Sources}

\begin{itemize}
    \item \textbf{Orbis:} Turkish manufacturing firms (2005--2024) for firm-year outcomes and controls
    \item \textbf{CEPII BACI HS1 and UNIDO ISDB4/INDSTAT4:} Chinese import penetration, output competition IV, input supply shocks
    \item \textbf{WIOD:} National input-output linkages
    \item \textbf{Turkstat:} PPI deflators and pre-2016 labor share by sector
\end{itemize}

\subsection{Chinese Import Penetration and Shift-Share Construction}

\subsubsection{Chinese Import Penetration}

Let $IP_{jt}$ denote import penetration from China into Turkey in industry $j$ and year $t$:
\[
IP_{jt} = \frac{M_{jt}^{\text{CHN} \to \text{TR}}}{Y_{jt} + M_{jt}^{\text{World} \to \text{TR}} - X_{jt}^{\text{TR} \to \text{World}}}
\]

and let $\Delta IP_{jt}$ denote the year-to-year change:
\[
\Delta IP_{jt} = IP_{jt} - IP_{jt-1}
\]

\paragraph{Endogeneity Concerns}

Estimating the impact of $\Delta IP_{jt}$ on firm outcomes is biased due to:
\begin{itemize}
    \item \textbf{Simultaneity:} Turkish industry-level demand shocks affect both imports from China and firm outcomes
    \item \textbf{Reverse causality:} Persistent industry characteristics jointly shape import exposure
    \item \textbf{Measurement error:} Firm outcomes are measured at firm level while exposure is at industry-year
\end{itemize}

To isolate exogenous variation, we construct an instrument based on Chinese export supply shocks rather than Turkey-specific demand.

\subsubsection{Shift-Share Output and Input Competition Shocks}

This construction adapts \cite{autorChinaSyndromeLocal2013}, and \cite{aghionOpposingFirmLevelResponses2024} to the Turkish context following the ``many exogenous shifts'' approach of \cite{borusyakQuasiExperimentalShiftShareResearch2022}.

\paragraph{Construction}

\begin{itemize}
    \item \textbf{Shift} $g_{jt}$: Growth in Chinese exports to high-income countries (EU15, USA, Japan, Australia) in industry $j$:
    \[
    \Delta \ln \text{Import Share}_{jt}^{\text{CHN} \to \text{HI}} = \Delta \ln M_{jt}^{\text{CHN} \to \text{HI}} - \Delta \ln M_{jt}^{\text{World} \to \text{HI}}
    \]

    \item \textbf{Share} $s_{j,\text{CHN}}^{t_0}$: Pre-determined share of Chinese import penetration to Turkey in baseline year $t_0$:
    \[
    s_{j,\text{CHN}}^{t_0} = \frac{M_{j,\text{CHN},t_0}}{Y_{jt_0} + M_{jt_0} - X_{jt_0}}
    \]

    \item \textbf{Direct Output Shock:}
    \[
    Z_{jt}^{\text{Output}} = s_{j,\text{CHN}}^{t_0} \times \Delta \ln M_{j,t}^{\text{CHN} \to \text{HI}}
    \]

    \item \textbf{Input Control \parencite{amitiTradeLiberalizationIntermediate2007}:}
    \[
    Z_{jt}^{\text{Input}} = \sum_k w_{jk}^{t_0} \times Z_{kt}^{\text{Output}}
    \]
    where $w_{jk}^{t_0}$ is the technical coefficient from the 2006 Turkish input-output table:
    \[
    w_{jk}^{t_0} = \frac{\text{Domestic Use}_{jk} + \text{Imported Use}_{jk}}{\text{Total Intermediate Inputs}_j}
    \]
\end{itemize}

\subsection{Markup Estimation}

\subsubsection{Setup}

I estimate firm markups using the production approach \parencite{deridderHitchhikersGuideMarkup2026,deloeckerMarkupsFirmLevelExport2012}. For firm $i$ in year $t$, define:
\begin{align*}
r_{it} &= \log R_{it} \quad \text{(log revenue)} \\
x_{it} &= \log X_{it} \quad \text{(log COGS)} \\
k_{it} &= \log K_{it} \quad \text{(log capital)}
\end{align*}

Sample restrictions: $R_{it} > 0$, $X_{it} > 0$, $K_{it} > 0$.

For a flexible (variable) input, the production approach gives:
\[
\mu_{it} = \alpha_{it}^{\text{COGS}} \cdot \frac{R_{it}}{X_{it}}
\]

In logs:
\[
\log \hat{\mu}_{it} = \log \hat{\alpha}_{it}^{\text{COGS}} + \log R_{it} - \log X_{it}
\]

\subsubsection{Estimating the Output Elasticity of COGS}

\paragraph{Cobb--Douglas (constant elasticity)}

Estimate within ISIC4:
\[
r_{it} = \beta_x x_{it} + \beta_k k_{it} + \lambda_t + \varepsilon_{it}
\]

Then $\hat{\alpha}_{it}^{\text{COGS}} = \hat{\beta}_x$.

\paragraph{Two-input Translog (elasticity varies with input levels)}

Estimate:
\[
r_{it} = \beta_x x_{it} + \beta_k k_{it} + \beta_{xx} x_{it}^2 + \beta_{kk} k_{it}^2 + \beta_{xk} x_{it} k_{it} + \lambda_t + \varepsilon_{it}
\]

Then the firm-year COGS elasticity is:
\[
\hat{\alpha}_{it}^{\text{COGS}} = \frac{\partial r_{it}}{\partial x_{it}} = \hat{\beta}_x + 2\hat{\beta}_{xx} x_{it} + \hat{\beta}_{xk} k_{it}
\]

\subsubsection{First Differences: Markup Adjustments}

\[
\Delta \log \hat{\mu}_{it} \equiv \log \hat{\mu}_{it} - \log \hat{\mu}_{i,t-1}
\]

\subsection{Labor Intensity}

Labor intensity is constructed from Turkish Statistical Institute Structural Business Statistics as the ratio of personnel costs to value added, averaged over 2009--2015 (pre-period) at the 4-digit sector level. This time-invariant measure captures differences in production technology across industries.

\subsection{Sector-Level Markup: Market-Share-Weighted Harmonic Mean}

Following \cite{bursteinBottomUpMarkupFluctuations2025}, define sector markup as the inverse harmonic mean of firm markups weighted by market share:
\[
\mu_{jt} = \left( \sum_{i \in j} s_{ijt} \mu_{it}^{-1} \right)^{-1}
\]

where $s_{ijt} = \frac{D_{it}}{\sum_{i' \in j} D_{i't}}$ is firm market share (domestic sales deflated by industry PPI).

The Herfindahl-Hirschman index is:
\[
\text{HHI}_{jt} = \sum_{i \in j} s_{ijt}^2
\]

Markup adjustment over time:
\[
\Delta \ln \mu_{jt} \equiv \ln \mu_{jt} - \ln \mu_{jt-1}
\]

\section{Empirical Strategy and Results}

\subsection{Baseline Model (OLS/IV)}

\subsubsection{First Stage}

\begin{equation}
\begin{aligned}
\Delta IP_{jt} &= \beta_1 Z_{jt}^{\text{Output}} + \beta_2 Z_{jt}^{\text{Input}} + \gamma_1 \text{LaborShare}_{jt_0} \times \text{Post2016}_t \\
&\quad + \gamma_2 \text{LaborShare}_{jt_0} + X'_{ijt-1} \gamma + \tau_t + \delta_j + u_{jt}
\end{aligned}
\tag{1}
\end{equation}

\subsubsection{Second Stage}

\begin{equation}
\begin{aligned}
Y_{ijt} &= \beta_O \widehat{\Delta IP_{jt}} + \beta_I Z_{jt}^{\text{Input}} + \gamma_1 \text{LaborShare}_{jt_0} \times \text{Post2016}_t \\
&\quad + \gamma_2 \text{LaborShare}_{jt_0} + X'_{ijt-1} \gamma + \tau_t + \delta_j + \varepsilon_{ijt}
\end{aligned}
\tag{2}
\end{equation}

where:
\begin{itemize}
    \item $Y_{ijt}$: Outcomes (exit at $t+1$ and markup adjustments)
    \item $\text{LaborShare}_{jt_0}$: Pre-2016 averaged labor intensity. The interaction with $\text{Post2016}_t$ captures differential effects after the 2016 minimum wage hike that evidently cause attrition of formal firms from ORBIS database.
    \item $X_{ijt-1}$: Lagged firm-year controls (age, size, export status, leverage, liquidity ratio)
\end{itemize}

\subsubsection{Exit Definition}

A firm is coded as exiting in year $t$ if observed in $t$ but not in $t+1, t+2$, or $t+3$. This three-year window reduces spurious exits from intermittent reporting. The last three years are right-censored.

\textbf{Prediction:} $\beta_O > 0$, indicating that selection is endogenous and related to the trade shock.

\subsubsection{Markup Adjustments}

\textbf{Prediction:} $\beta_O < 0$, reflecting pro-competitive effects from import competition.

\subsubsection{Assumptions}

\begin{itemize}
    \item \textbf{Relevance:} Chinese export surge to high-income economies affects import penetration in Turkey
    \item \textbf{Exclusion restriction:} Chinese export boom in high-income economies does not directly impact Turkish firm survival or pricing power
\end{itemize}

\subsection{Selection Correction}

To correct selection bias from firm exit, I use a semiparametric control function motivated by \autocite{backusWhyProductivityCorrelated2020}, which is based on a Markov structure: there exists an unobserved firm state $\theta_{it}$ such that survival follows a threshold in $\theta_{it-1}$, and the predictable component of $\Delta \ln \mu_{ijt}$ is a function of $\theta_{it-1}$.

Formally:
\[
\text{Observe } \Delta \ln \mu_{ijt} \Longleftrightarrow \text{survive to } t \text{ and observed at } t-1
\]

\[
\text{Survive at } t \Longleftrightarrow \theta_{it-1} \geq \theta_i^*(S_{t-1})
\]

where $S_{t-1}$ includes sector-year conditions and import penetration.

\subsubsection{Control Function Approach}

Use lagged EBIT margin as a state proxy $s_{it-1}$ that predicts both survival and markup dynamics. Assume:
\[
\mathbb{E}[\varepsilon_{it} | s_{it-1}, \delta_t, \tau_t] = \mathbb{E}[\varepsilon_{it} | \theta_{it-1}, \delta_t, \tau_t]
\]

The error term decomposes as:
\[
\varepsilon_{i(j)t} = \underbrace{\lambda_{jt}}_{\text{sector-year shock}} + \underbrace{\eta_{it}}_{\text{idiosyncratic shock}}
\]

Among survivors:
\[
\mathbb{E}[\eta_{it} | \theta_{it-1} \geq \theta_i^*(S_{t-1})] = \mathbb{E}[\eta_{it} | s_{it-1}] = g(s_{it-1})
\]

where $g(\cdot)$ is approximated with a third-order polynomial series.

The corrected second stage becomes:
\begin{equation}
\begin{aligned}
\Delta \ln \mu_{i(j)t} &= \beta_O \widehat{\Delta IP_{jt}} + \beta_I Z_{jt}^{\text{Input}} + \gamma_1 \text{LaborShare}_{jt_0} \times \text{Post2016}_t \\
&\quad + \gamma_2 \text{LaborShare}_{jt_0} + X'_{ijt-1} \gamma + \tau_t + \delta_j + g(s_{it-1}) + \epsilon_{i(j)t}
\end{aligned}
\tag{3}
\end{equation}

\subsubsection{Initial Results}
    \begin{itemize}
        \item Baseline IV: Chinese import competition decreases average firm markup adjustments. However, the shocks also increase exit probability, which introduces selection bias in estimating markup adjustments. 
        \item Selection correction: The control function methodology using lagged EBIT margin as state proxy is validated in Table 3 (Empirical Appendix). After controlling for selection, the negative effect on markup adjustments remains significant, suggesting that surviving firms reduce markups in response to import competition. The selection bias can be (suggestively) quantified by comparing the coefficients in column (3) and (4) in the Table 1. 
        \item The input supply shock does not have a significant effect on markup adjustments, indicating that the direct output competition channel is more relevant for pricing behavior than input cost changes. 
    \end{itemize}

\subsection{Grouped IV Quantiles}

\subsubsection{Identification}

To assess heterogeneous effects across the markup adjustment distribution, I use grouped IV quantile regression (\cite{chetverikovIVQuantileRegression2016}). Let $\rho_{jt}^{(k)}$ be the $k$-th decile of the industry-level distribution of $\Delta \ln \mu_{it}$ among survivors:

\begin{equation}
\begin{aligned}
\rho_{jt}^{(k)} &= \beta_O^{(k)} \widehat{\Delta IP_{jt}} + \beta_I^{(k)} Z_{jt}^{\text{Input}} + \gamma_1^{(k)} \text{LaborShare}_{jt_0} \times \text{Post2016}_t \\
&\quad + \gamma_2^{(k)} \text{LaborShare}_{jt_0} + X'_{jt-1} \gamma^{(k)} + \tau_t^{(k)} + \delta_j^{(k)} + \nu_{jt}^{(k)}
\end{aligned}
\tag{4}
\end{equation}

Identification requires:
\begin{itemize}
    \item Relevance: Output competition shifts affect import competition
    \item Exclusion restriction: Conditional on fixed effects and controls,
    \[
    \mathbb{E}\Big[\nu_{jt}^{(k)} \mid Z_{jt}^{\text{Output}}, \delta_j, \tau_t, \text{controls}\Big] = 0
    \]
\end{itemize}

\subsubsection{Order Statistic Bias Correction}

In small sectors with fewer firms than quantiles, I add a control function to correct order statistic bias:

\begin{equation}
\begin{aligned}
\rho_{jt}^{(k)} &= \beta_O^{(k)} \widehat{\Delta IP_{jt}} + \beta_I^{(k)} Z_{jt}^{\text{Input}} + \gamma_1^{(k)} \text{LaborShare}_{jt_0} \times \text{Post2016}_t \\
&\quad + \gamma_2^{(k)} \text{LaborShare}_{jt_0} + X'_{jt-1} \gamma^{(k)} + \tau_t^{(k)} + \delta_j^{(k)} + g^{(k)}(n_{jt}) + \nu_{jt}^{(k)}
\end{aligned}
\tag{5}
\end{equation}

where $g^{(k)}(\cdot)$ is approximated with a third-order polynomial in the number of firm observations $n_{jt}$.

\subsubsection{Heterogeneity}

To test whether effects differ by market structure, I interact the import penetration shock with lagged domestic concentration (HHI/CR10/CR4):

\begin{equation}
\begin{aligned}
\rho_{jt}^{(k)} &= \beta_{O1}^{(k)} \widehat{\Delta IP_{jt}} + \beta_{O2}^{(k)} \widehat{\Delta IP_{jt}} \times \text{Concentration}_{jt-1} + \beta_I^{(k)} Z_{jt}^{\text{Input}}  + \gamma_2^{(k)} \text{LaborShare}_{jt_0}\\
&\quad + \gamma_1^{(k)} \text{LaborShare}_{jt_0} \times \text{Post2016}_t + X'_{jt-1} \gamma^{(k)} + \tau_t^{(k)} + \delta_j^{(k)} + g^{(k)}(n_{jt}) + \nu_{jt}^{(k)}
\end{aligned}
\tag{6}
\end{equation}

where $g^{(k)}(\cdot)$ is approximated with a third-order polynomial in the number of firm observations $n_{jt}$.

\subsubsection{Initial Results}
    \begin{itemize}
        \item Grouped IV quantile estimates reveal near-zero effects across central distribution (10th-70th percentiles), but increasingly negative coefficients from 75th percentile onward, reaching roughly $-2.0$ at the 90th percentile. This suggests that units located at the top of the sector-year markup-adjustment distribution experience stronger pro-competitive effects from import competition. The within-firm discipline mechanism operates primarily through constraining large positive adjustments among high-markup firms, while leaving the central distribution largely unaffected (Figure 1).
        \item The compression is amplified where domestic market structure is more concentrated (high CR4), beyond compositional effects from selection (Figure 2). This is consistent with a mechanism where import competition constrains the pricing power of high-markup firms, but has limited impact on the central distribution of firms that are already operating with low markups. In more concentrated sectors, the top of the distribution is more likely to be occupied by high-markup firms that are vulnerable to import competition, leading to stronger pro-competitive effects at the upper tail.

\subsection{Decomposition Approach}

To understand mechanisms, I decompose changes in sector markups following \cite{bursteinBottomUpMarkupFluctuations2025}.

\subsubsection{Decomposition}

From the harmonic mean sector markup:
\[
\mu_{jt}^{-1} = \sum_{i \in j} s_{ijt} \mu_{it}^{-1}
\]

Decompose year-to-year change in inverse sector markup:
\begin{equation}
\Delta \mu_{jt}^{-1} \approx \underbrace{\sum_{i \in j} \bar{s}_{ij} \Delta \mu_{it}^{-1}}_{\text{within-firm adjustment}} + \underbrace{\sum_{i \in j} \bar{\mu}_{ij}^{-1} \Delta s_{it}}_{\text{reallocation of shares}} \tag{7}
\end{equation}

where $\bar{s}_{ij} = \frac{1}{2}(s_{ijt} + s_{ijt-1})$ and $\bar{\mu}_{ij}^{-1} = \frac{1}{2}(\mu_{it}^{-1} + \mu_{it-1}^{-1})$.

Following \textcite{melitzDynamicOlleyPakesProductivity2015}, 
let $S_t$ denote the set of firms present at time $t$. 
Define continuers $C = S_{t-1} \cap S_t$, 
entrants $E = S_t \setminus S_{t-1}$, 
and exiters $X = S_{t-1} \setminus S_t$. 
I decompose the change in inverse sector markup as follows:

\[
\Delta \mu_{jt}^{-1}
=
\underbrace{\sum_{i \in C} \bar{s}_{ij}\,\Delta \mu_{ij}^{-1}}_{\text{Within (continuers)}}
+
\underbrace{\sum_{i \in C} \Delta s_{ij}\,\bar{\mu}_{ij}^{-1}}_{\text{Between (continuers)}}
+
\underbrace{\sum_{i \in E} s_{ijt}\,\mu_{ijt}^{-1}}_{\text{Entry}}
-
\underbrace{\sum_{i \in X} s_{ij,t-1}\,\mu_{ij,t-1}^{-1}}_{\text{Exit}} .
\]


\subsubsection{Relative Importance}

\[
\text{Within Share}_{jt} = \frac{W_{jt}}{W_{jt} + R_{jt}}, \quad \text{Reallocation Share}_{jt} = \frac{R_{jt}}{W_{jt} + R_{jt}}
\]

\begin{itemize}
    \item Output competition: Incumbent within-firm adjustment pushes inverse markups up (reducing markups - pro-competitive), while reallocation pushes inverse markups down (anti-competitive), but neither component is precisely estimated.
    \item Input supply shocks: Within-firm inverse markup adjustment is negative, while reallocation is positive. Coefficients are statistically significant at 5\% level, suggesting that input supply shocks lead to anti-competitive within-firm adjustments but pro-competitive reallocation effects. Cost-shock pass-through dynamics. 
    \item Output competition is strong for incumbents (firm-level discipline), but washed out in sector aggregates due to selection and reallocation effects. Whereas, input supply shocks are weak at the firm level but amplify at the sector level through reallocation, consistent with a cost channel that affects all firms and leads to market share shifts toward more efficient producers. 
\end{itemize}

\section{Robustness and Sensitivity Analysis}

\subsection{Identification Robustness}

\begin{itemize}
    \item Instrument validity diagnostics: \autocite{adaoShiftShareDesignsTheory2019}, Rotemberg weights concentration tests
    \item Alternative base periods and exposure definitions (absorption-based vs. import/value-added)
    \item First-stage diagnostics: Kleibergen-Paap F-statistic, Anderson-Rubin inference, wild cluster bootstrap
    \item Alternative fixed effects specifications (firm FE, sector FE, sector $\times$ linear trend)
    \item Two-way clustering (sector and year)
\end{itemize}

\subsection{Event Study and Placebo Tests}

\begin{itemize}
    \item Using output competition shock variable at time $t+1, t+2, t+3$ as falsification test whether future shocks predict current outcomes. The figure 3 shows that the coefficients are small, inconsistent in sign, not statistically significant, non-monotonic, which provides evidence against reverse causality or pre-trends driving the results.

\begin{itemize}
    \item Alternative specifications: GMM, Cobb-Douglas
    \item Alternative trimming/winsorization of extreme markup adjustments
    \item Alternative pricing power proxies: price-cost margin, gross margin
    \item Robustness using revenue-side markup estimation (Kirov, Mengano, and Traina, 2018)

\begin{itemize}
    \item Balanced vs. unbalanced panels
    \item Excluding micro firms and top 1\% size
    \item Alternative manufacturing groupings (2-digit, 3-digit)
\end{itemize}

\subsection{Selection Correction Robustness}

    \item Alternative state proxies: lagged size, lagged markup level
    \item Polynomial order sensitivity (binning as nonparametric alternative)
    \item Lee (2009) bounds as external validity check
\end{itemize}

\begin{itemize}
    \item Alternative quantile choices (quintiles vs. deciles)
    \item Alternative ranking variables
    \item Bootstrap standard errors at sector level
    \item Excluding sectors with few firms per year
\end{itemize}

\subsection{Mechanism Checks}

\begin{itemize}
    \item Industry concentration measures: HHI, CR10, CR4
    \item Do large incumbents gain share when import competition rises? 
    \[
    \Delta \ln (share_{ijt}) = \beta \Delta IP_{jt-1} \times H_{ijt-1} + \gamma X'_{ijt-1} + \alpha_i + \delta_{jt} + \varepsilon{ijt}
    \]
    where $H_{ijt-1}$ can be lagged firm size, log sales, CR4 membership. A positive $\beta$ would suggest that larger incumbents gain share when import competition rises. This support strategic rivalry and dominant-firm reallocation. 
    \item Regress EBIT margin/gross margin on import competition to test for cost pass-through. If import competition reduces markups through a cost channel, we should see a significant negative effect on margins. If the effect is primarily through competitive discipline, the margin effect may be smaller or insignificant.
\end{itemize}

\section{Conclusions}

\subsection{Key Findings}

Import competition reduces markup dispersion through a ``catch-up discipline'' mechanism: firms attempting large positive markup adjustments (75th--90th percentiles) face coefficients approaching $-2.0$, while central distribution firms experience near-zero effects. This pro-competitive discipline is strongest in concentrated sectors, where high-markup firms are most vulnerable to foreign competition.

The decomposition reveals asymmetric channels: output shocks operate via within-firm discipline on survivors, while input shocks trigger pro-competitive reallocation toward more efficient producers. Selection bias does not reverse the sign of effects.

\subsection{Implications}

Trade openness disciplines pricing power precisely where domestic concentration was strongest. This suggests trade liberalization and domestic competition policy are substitutable mechanisms for controlling market power, with important distributional consequences for frontier versus marginal firms.

\subsection{Limitations}

Turkey's benefits from EU integration may cushion effects from Chinese import penetration, potentially understating the true magnitude of competition effects.

\section{Theoretical Appendix}

\subsection{Benchmark setup with constant markups}
Assume a representative consumer has Cobb-Douglas preferences over industry-level aggregates $Q_{jt}$ of quantities consumed ($q_{ijt}$) of all varieties available in industry j at time t: 
\[
U_t = \prod_{j\in J} Q_{jt}^{\alpha_j}, \quad \sum_j \alpha_j = 1
\]
and nested CES preferences over varieties within each industry, with domestic and foreign products treated symmetrically at the level of substitution across varieties: 
\[
Q_{jt} = \Bigg( \sum_{h \in \Omega_{jt}} q_{hjt}^{\frac{\gamma-1}{\gamma}} \Bigg)^{\frac{\gamma}{\gamma-1}}, \quad \gamma>1
\]
Here $h$ indexes individual varieties produced either by a domestic firm i or by a foreign supplier (including China); $\Omega_{jt}$ is the set of varieties available in industry j and year t. 

Let $P_{jt}$ denote the \textit{ideal} CES price index in industry j: $$P_{jt} = \Bigg( \sum_{h \in \Omega_{jt}} p_{hjt}^{1-\gamma} \Bigg)^{\frac{1}{1-\gamma}}$$ and let $X_{jt}$ be total expenditure on industry j, we have $X_{jt}=P_{jt}Q_{jt}$. Differentiating the expenditure minimization problem: 
\[
\min \sum_h p_{hjt}q_{hjt} \quad \text{such that} \quad Q_{jt} = \Bigg( \sum_{h \in \Omega_{jt}} q_{hjt}^{\frac{\gamma-1}{\gamma}} \Bigg)^{\frac{\gamma}{\gamma-1}}, \quad \gamma>1,
\]
gives the standard CES result: $$s_{hjt} = \frac{p_{hjt}q_{hjt}}{X_{jt}}= \frac{p_{hjt}^{1-\gamma}}{P_{jt}^{1-\gamma}}$$and expenditure shares at the country level are: $s_{cjt} \equiv \sum_{h \in \Omega_{cjt}}s_{hjt}$ where $c \in \{TR, CHN, RoW \}$ indexes origin country and $\Omega_{cjt}$ is the set of varieties from origin c in industry j at time t. Import penetration from China in industry j can be written as $IP^{CHN}_{jt} \approx s_{CHN,jt}$, which means the Chinese expenditure share in industry j. In the data, I approximate this share using absorption-based import penetration.

Let assume a domestic firm $i$ industry $j$ has a production technology: $y_{ijt}= A_{jt}\varphi_{it}l_{ijt}$ where $A_{jt}$ is industry-time specific technology, $\varphi_{ij}$ is a time-invariant firm-specific productivity draw, and $l_{ijt}$ is labor input. Variable cost is $\frac{w_ty_{ijt}}{A_{jt}\varphi_{ij}}$ with a common wage $w_t$. Let $f_{jt}$ denote the fixed operating cost for firms at industry $j$, time $t$ in labor units to remain active. Foreign firms (including Chinese producers) have analogous technologies with productivity draws distributed according to country-specific productivity distributions and facing trade costs into Turkey. Denote $\mu_{ijt}$ as the gross price-cost markup: $\mu_{ijt} \equiv \frac{p_{ijt}}{mc_{ijt}}$. Under monopolistic competition with CES demand, $\mu_{ijt} = \frac{\gamma}{\gamma-1}$ is constant and independent of the intensity of import competition. 

The benchmark model implies that standard CES models with monopolistic competition cannot generate the strong, negative response of firm-level markups to changes in Chinese import penetration. Therefore, an oligopolistic competition structure with variable markups is necessary to rationalize the empirical findings. 

\subsection{Oligopoly wedge: strategic pricing and variable markups}

Following the approach in \textcite{edmondCompetitionMarkupsGains2015} and \textcite{amitiInternationalShocksVariable2019}, consider a general oligopolistic environment within each industry $j$ in which firms face the CES demand system described above but recognize their influence on the industry price index.

Define a strategic pricing intensity (own-price markup elasticity): $$\Gamma_{ijt} \equiv - \frac{\partial \log \mu_{ijt}}{\partial \log p_{ijt}}$$with $\Gamma_{ijt} = 0$ under constant markup monopolistic competition and $\Gamma>0$ under oligopoly/strategic interactions, where prices and markups comove with market shares and rivals' responses.

A convenient way to summarize pricing behavior is through pass-through of marginal cost into prices. For small changes in own marginal cost $mc_{hjt}$: 
\[
\frac{\partial \log p_{hjt}}{\partial \log mc_{hjt}} = \frac{1}{1+\Gamma_{hjt}}
\]

so the strength of markup adjustment is governed by $\Gamma_{hjt}$. When competition intensifies due to entry or expansion of foreign producers, strategic interactions imply that domestic firms optimally reduce their markups.

Sector-level markups are defined as a market-share-weighted harmonic mean: 
\[
\mu_{jt} = \bigg(\sum_{i\in j} s_{ijt}\mu_{ijt}^{-1} \bigg)^{-1}
\]
where $s_{ijt}$ is the domestic sales share of firm $i$ in industry $j$ at time $t$. This aggregator arises in oligopolistic models where price-cost margins at the sector-level depend on firms' market shares and markups. 

The optimal log markups of firm $i$ at industry $j$ depends on its competitive environment, including market share and the structure of competition in that industry: $\log \mu_{ijt} = \psi(s_{ijt}, s_{-ijt}; \Theta_{j})$ where $s_{-ijt}$ is the shares of rivals (domestic and foreign) and $\Theta_j$ are industry-specific demand and cost primitives. A small change in Chinese penetration can be written as a shift in the distribution of foreign market shares within industry j: $\Delta IP^{CHN}_{jt} \approx \sum_{h\in \Omega_{CHN,jt}} \Delta s_{hjt}$ where $h$ is an individual variety index, and $\Omega_{CHN,jt}$ is the set of Chinese varieties available in industry j at time $t$.

Using the implicit function of markups, the total derivative of $\mu_{ijt}$ with respect to $IP^{CHN}_{jt}$ is: 
\[
\frac{\partial \log \mu_{ijt}}{\partial IP_{jt}^{CHN}} = \sum_{h \in \Omega_{CHN,jt}} \bigg(\frac{\partial \log \mu_{ijt}}{\partial s_{ijt}} \frac{\partial s_{ijt}}{\partial s_{hjt}} + \sum_{k \neq i} \frac{\partial \log \mu_{ijt}}{\partial s_{kjt}} \frac{\partial \log \mu_{kjt}}{\partial s_{hjt}} \bigg)
\]
Under standard oligopolistic structures (Cournot/Bertrand with CES demand), we can see that when competition from Chinese firms increases, domestic market shares fall and markups decline. 

Define the industry-average markup response as a weighted share over firms $i \in j$:
\[
\bar{\beta}_j \equiv \sum_{i \in j} \varpi_{ij} \frac{\partial \log \mu_{ijt}}{\partial IP_{jt}^{CHN}}
\]

where $\varpi_{ij}$ are within-industry weights. In a standard panel regression with sector fixed effects, these weights are determined by how much variation firm $i$ contributes to the within-industry variation in the outcome. Intuitively, firms that experience large markup changes have more weight in recovering the average slope of the regression line within industry $j$.

Following \textcite{EmpiricalStrategiesLabor1999}, the OLS/IV estimator with heterogeneous treatment effects recovers a weighted average of industry-level slopes $\bar{\beta}_j$: 
\[
\beta_{\mu} = \int_j w_j \bar{\beta}_j dj = \int_j w_j \bigg(\sum_{i \in j} \varpi_{ij} \frac{\partial \log \mu_{ijt}}{\partial IP_{jt}^{CHN}} \bigg)dj
\]

where $w_j \geq 0$ are across-industry weights with $\int_j w_j d_j = 1$. In the IV design with sector and year fixed effects, $w_j$ is proportional to the first-stage variance of the instrument $Z^{Output}_{jt}$ within industry $j$. Industries where the shift-share instrument predicts more variation in Chinese import penetration receive higher weight. Since all the structural derivative terms in the double sum are negative under oligopolistic competition, the weighted share average is also negative.

\end{enumerate}


  \item \textbf{Catch-up discipline mechanism:} Low-margin firms attempting large positive adjustments are constrained, while high-markup firms remain unconstrained (they were not attempting large increases). This reinforces rather than offsets average pro-competitive effects.
